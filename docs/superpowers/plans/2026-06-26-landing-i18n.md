# 落地页多语言（语言标注 + hreflang + RTL）实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 让每张落地页可标注语言（驱动 `lang`/`dir`/OG locale）并可选声明其它语言版本（hreflang 互链），RTL 语言用 `dir="rtl"`。

**Tech Stack:** Next.js 16(App Router)、TypeScript、Tailwind、vitest、Playwright。

```ts
// lib/i18n/lang.ts
// 落地页语言工具：RTL 判定 / 规范化 / hreflang 映射 / 语言下拉选项。
import type { PageI18n } from "@/types/schema.draft";

const RTL_PREFIXES = ["ar", "he", "fa", "ur", "ps", "sd", "ug", "yi"];

/** 取主语言子标签（ar-EG → ar），前缀匹配 RTL 集。 */
export function isRtlLang(lang?: string): boolean {
  if (!lang) return false;
  const primary = lang.trim().toLowerCase().split("-")[0];
  return RTL_PREFIXES.includes(primary);
}

export function normalizeLang(lang?: string): string | undefined {
  const t = lang?.trim().toLowerCase();
  return t ? t : undefined;
}

/** alternates → Next Metadata.alternates.languages 映射（+ x-default=canonical）。无有效项返回 undefined。 */
export function resolveHreflang(i18n: PageI18n | undefined, canonical: string): Record<string, string> | undefined {
  const alts = (i18n?.alternates ?? []).filter((a) => a.lang.trim() && a.url.trim());
  if (alts.length === 0) return undefined;
  const map: Record<string, string> = {};
  for (const a of alts) map[a.lang.trim()] = a.url.trim();
  map["x-default"] = canonical;
  return map;
}

/** 常用投放语言（code + 中文名）。 */
export const LANG_OPTIONS: { code: string; label: string }[] = [
  { code: "en", label: "英语 English" },
  { code: "zh", label: "中文" },
  { code: "es", label: "西班牙语 Español" },
  { code: "pt", label: "葡萄牙语 Português" },
  { code: "fr", label: "法语 Français" },
  { code: "de", label: "德语 Deutsch" },
  { code: "it", label: "意大利语 Italiano" },
  { code: "ru", label: "俄语 Русский" },
  { code: "ar", label: "阿拉伯语 العربية" },
  { code: "tr", label: "土耳其语 Türkçe" },
  { code: "ja", label: "日语 日本語" },
  { code: "ko", label: "韩语 한국어" },
  { code: "id", label: "印尼语 Indonesia" },
  { code: "th", label: "泰语 ไทย" },
  { code: "vi", label: "越南语 Tiếng Việt" },
  { code: "he", label: "希伯来语 עברית" },
];
```

```tsx
"use client";
// landing-editor/forms/I18nForm.tsx
// 页面语言：语言代码（驱动 lang/dir/OG locale）+ 可选其它语言版本（hreflang）。
import type { PageI18n, AlternateLocale } from "@/types/schema.draft";
import { LANG_OPTIONS } from "@/lib/i18n/lang";
import { Field } from "../ui/Field";
import { Select } from "../ui/Select";
import { TextInput } from "../ui/TextInput";
import { RepeatableList } from "../ui/RepeatableList";

export function I18nForm({ value, onChange }: { value: PageI18n; onChange: (v: PageI18n) => void }) {
  return (
    <div className="space-y-3">
      <Field label="页面语言（影响 lang / 阅读方向 / OG locale）">
        <Select value={value.lang} onChange={(e) => onChange({ ...value, lang: e.target.value })}>
          {LANG_OPTIONS.map((o) => <option key={o.code} value={o.code}>{o.label}（{o.code}）</option>)}
        </Select>
      </Field>
      <RepeatableList<AlternateLocale>
        label="其它语言版本（可选，用于 hreflang 互链）"
        items={value.alternates ?? []}
        onChange={(alternates) => onChange({ ...value, alternates: alternates.length ? alternates : undefined })}
        create={() => ({ lang: "en", url: "" })}
        addLabel="添加语言版本"
        renderItem={(item, onItemChange) => (
          <div className="grid grid-cols-[8rem_1fr] gap-2">
            <Select value={item.lang} onChange={(e) => onItemChange({ ...item, lang: e.target.value })}>
              {LANG_OPTIONS.map((o) => <option key={o.code} value={o.code}>{o.code}</option>)}
            </Select>
            <TextInput value={item.url} onChange={(e) => onItemChange({ ...item, url: e.target.value })} placeholder="https://… 该语言版本 URL" />
          </div>
        )}
      />
    </div>
  );
}
```

```ts
// e2e/i18n.spec.ts
// 多语言：已发布页带 i18n{lang:ar, alternates} → 自有域公开页 HTML 含 dir=rtl / lang=ar / hreflang。
import { test, expect, request as playwrightRequest } from "@playwright/test";
import { Pool } from "pg";
import { config as loadEnv } from "dotenv";
import { E2E_USER_ID, SLUG_PREFIX, isDbE2EEnabled } from "./helpers/db";
import { skincareConsultDraft } from "@/landing-editor/samples/skincareConsultDraft";

loadEnv({ path: ".env.local" });
loadEnv({ path: ".env" });

const SLUG = `${SLUG_PREFIX}i18n`;
const CUSTOM_DOMAIN = "e2e-i18n.test";

let pool: Pool;
let landingPageId: string;

function makePool(): Pool {
  const cs = process.env.E2E_DATABASE_URL ?? process.env.DATABASE_URL ?? "";
  const isLocal = cs.includes("localhost") || cs.includes("127.0.0.1");
  return new Pool({ connectionString: cs, ssl: isLocal ? false : { rejectUnauthorized: false } });
}

const draftWithI18n = {
  ...skincareConsultDraft,
  i18n: { lang: "ar", alternates: [{ lang: "en", url: "https://en.example.com/" }] },
};

test.describe("落地页多语言（自有域）", () => {
  test.skip(!isDbE2EEnabled, "Set RUN_DB_E2E=1 to run database-backed e2e tests.");

  test.beforeAll(async () => {
    pool = makePool();
    await pool.query(`DELETE FROM domains WHERE domain = $1`, [CUSTOM_DOMAIN]);
    await pool.query(`DELETE FROM landing_pages WHERE slug = $1`, [SLUG]);
    const lp = await pool.query(
      `INSERT INTO landing_pages (user_id, name, slug, status, data, published_at)
       VALUES ($1, $2, $3, 'published', $4::jsonb, NOW()) RETURNING id`,
      [E2E_USER_ID, SLUG, SLUG, JSON.stringify(draftWithI18n)],
    );
    landingPageId = lp.rows[0].id;
    await pool.query(
      `INSERT INTO domains (user_id, domain, enabled, verified, landing_page_id)
       VALUES ($1, $2, true, true, $3)`,
      [E2E_USER_ID, CUSTOM_DOMAIN, landingPageId],
    );
  });

  test.afterAll(async () => {
    if (pool) {
      await pool.query(`DELETE FROM domains WHERE domain = $1`, [CUSTOM_DOMAIN]);
      await pool.query(`DELETE FROM landing_pages WHERE slug = $1`, [SLUG]);
      await pool.end();
    }
  });

  test("RTL 页 head/根含 dir=rtl + lang=ar + hreflang", async () => {
    const ctx = await playwrightRequest.newContext();
    try {
      const res = await ctx.get("http://localhost:3001/", { headers: { Host: CUSTOM_DOMAIN } });
      expect(res.status()).toBe(200);
      const html = await res.text();
      expect(html).toMatch(/dir="rtl"/);
      expect(html).toMatch(/lang="ar"/);
      expect(html).toMatch(/hreflang="en"/);
      expect(html).toMatch(/hreflang="x-default"/);
    } finally {
      await ctx.dispose();
    }
  });
});
```
